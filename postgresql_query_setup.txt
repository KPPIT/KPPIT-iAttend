# Create new table
CREATE TABLE union_members (
    staff_id INT,
    employee_name VARCHAR (255),
    company_name VARCHAR (255),
    organizational_unit VARCHAR (255),
    attendance VARCHAR (10),
    checkin_time TIMESTAMP
);


# View table
SELECT * FROM union_members;
